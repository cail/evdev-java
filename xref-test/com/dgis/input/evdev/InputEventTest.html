17  package com.dgis.input.evdev;
18  
19  import org.junit.Test;
20  
21  import java.io.IOException;
22  import java.nio.ShortBuffer;
23  
24  import static org.assertj.core.api.Assertions.assertThat;
25  
26  public class InputEventTest {
27  
28      @Test
29      public void testParsingSimpleValue() throws IOException {
30          ShortBuffer buffer = ShortBuffer.allocate(12);
31          // time seconds
32          write(buffer, 1, 2, 0, 0);
33          // time micro seconds
34          write(buffer, 4, 5, 0, 0);
35          // type
36          buffer.put((short) 8);
37          // code
38          buffer.put((short) 9);
39          // value
40          write(buffer, 10, 11);
41  
42          buffer.rewind();
43  
44          InputEvent event = InputEvent.parse(buffer, "source", "arch");
45          assertThat(event.source).isEqualTo("source");
46          assertThat(event.timeSec).isEqualTo((2L << 16) + 1);
47          assertThat(event.timeMicroSec).isEqualTo((5L << 16) + 4);
48          assertThat(event.type).isEqualTo((short)8);
49          assertThat(event.code).isEqualTo((short)9);
50          assertThat(event.value).isEqualTo(10 + (11 << 16));
51      }
52  
53      private void write(ShortBuffer buffer, int s0, int s1, int s2, int s3) {
54          buffer.put((short) s0);
55          buffer.put((short) s1);
56          buffer.put((short) s2);
57          buffer.put((short) s3);
58      }
59      private void write(ShortBuffer buffer, int s0, int s1) {
60          buffer.put((short) s0);
61          buffer.put((short) s1);
62      }
63  
64  }

777     /**
778      * Parse an InputEvent out of a ShortBuffer.
779      *
780      * @param shortBuffer
781      * @param source
782      * @return the parsed InputEvent
783      * @throws IOException
784      */
785     public static InputEvent parse(ShortBuffer shortBuffer, String source) throws IOException {
786         return parse(shortBuffer, source, null);
787     }
788 
789     /**
790      * Parse an InputEvent out of a ShortBuffer.
791      *
792      * @param shortBuffer
793      * @param source
794      * @param arch
795      * @return the parsed InputEvent
796      * @throws IOException
797      */
798     public static InputEvent parse(ShortBuffer shortBuffer, String source, String arch) throws IOException {
799         InputEvent e = new InputEvent();
800         short a, b, c, d;
801         a = shortBuffer.get();
802         b = shortBuffer.get();
803         if (arch.equals("arm")) {
804             e.time_sec = (b << 16) | a;
805         } else {
806             c = shortBuffer.get();
807             d = shortBuffer.get();
808             e.time_sec = (d << 48) | (c << 32) | (b << 16) | a;
809         }
810         a = shortBuffer.get();
811         b = shortBuffer.get();
812         if (arch.equals("arm")) {
813             e.time_usec = (b << 16) | a;
814         } else {
815             c = shortBuffer.get();
816             d = shortBuffer.get();
817             e.time_usec = (d << 48) | (c << 32) | (b << 16) | a;
818         }
819         e.type = shortBuffer.get();
820         e.code = shortBuffer.get();
821         c = shortBuffer.get();
822         d = shortBuffer.get();
823         e.value = (d << 16) | c;
824         e.source = source;
825         return e;
826     }
